CREATE OR REPLACE PROCEDURE AUTH_DB2_ZOS_REPORT.USP_ORGUNIT_PARA ( IN P_DB_CMDB_NAME VARCHAR(100) )
  SPECIFIC USP_ORGUNIT_PARA
  DYNAMIC RESULT SETS 1
  LANGUAGE SQL
  NOT DETERMINISTIC
  EXTERNAL ACTION
  MODIFIES SQL DATA
  CALLED ON NULL INPUT
  INHERIT SPECIAL REGISTERS
  OLD SAVEPOINT LEVEL
BEGIN 
  DECLARE cur CURSOR WITH RETURN FOR
 	SELECT DISTINCT RPAD(s.zust_orga,43) ORGUNIT
	FROM AUTH_DB2_ZOS_REPORT.VW_REPORT_DB2_ZOS_TABAUTH s
	WHERE TRIM(s.SYSPLEX_CMDB_NAME)=TRIM(P_DB_CMDB_NAME)
	ORDER BY 1;
		
	OPEN cur;
  -- Optional: exception expressions
  -- EXCEPTION 
  -- WHEN { exception-condition } THEN { statement; } ... { statement; } 
  -- ... 
  -- WHEN { exception-condition } THEN { statement; } ... { statement; } 
END
