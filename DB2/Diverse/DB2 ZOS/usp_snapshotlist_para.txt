CREATE OR REPLACE PROCEDURE AUTH_DB2_ZOS_REPORT.USP_SNAPSHOTLIST_PARA ( IN P_DB_CMDB_NAME VARCHAR(100), IN P_ZUST_ORGA VARCHAR(50), IN P_SACHGEBIET VARCHAR(50) )
  SPECIFIC USP_SNAPSHOTLIST_PARA
  DYNAMIC RESULT SETS 1
  LANGUAGE SQL
  NOT DETERMINISTIC
  EXTERNAL ACTION
  MODIFIES SQL DATA
  CALLED ON NULL INPUT
  INHERIT SPECIAL REGISTERS
  OLD SAVEPOINT LEVEL
BEGIN 
  DECLARE cur CURSOR WITH RETURN FOR
	SELECT DISTINCT RPAD(VARCHAR_FORMAT(s.SNAP_TS,'DD.MM.YYYY HH24:MI:SS'),35) SNAPSHOT
	FROM AUTH_DB2_ZOS_REPORT.VW_REPORT_DB2_ZOS_TABAUTH s
	WHERE TRIM(s.SYSPLEX_CMDB_NAME)=TRIM(P_DB_CMDB_NAME)
		AND TRIM(s.zust_orga)=TRIM(P_ZUST_ORGA) 
		AND TRIM(s.sachgebiet)=TRIM(P_SACHGEBIET)
	GROUP BY VARCHAR_FORMAT(s.SNAP_TS,'DD.MM.YYYY HH24:MI:SS');
		
	OPEN cur;
  -- Optional: exception expressions
  -- EXCEPTION 
  -- WHEN { exception-condition } THEN { statement; } ... { statement; } 
  -- ... 
  -- WHEN { exception-condition } THEN { statement; } ... { statement; } 
END
